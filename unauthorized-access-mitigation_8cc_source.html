134void SecureSink::HandleRead(Ptr<Socket> socket) {
135 while (Ptr<Packet> pkt = socket->Recv()) {
136 uint32_t size = pkt->GetSize();
137 std::vector<uint8_t> buffer(size);
138 pkt->CopyData(buffer.data(), size);
139 std::string data(reinterpret_cast<char*>(buffer.data()), size);
140
141 if (data.find("ID:") == 0) {
142 try {
143 uint32_t nodeId = std::stoi(data.substr(3, data.find("|ROLE:") - 3));
144 std::string role = data.substr(data.find("|ROLE:") + 6, data.find("|EXP:") - data.find("|ROLE:") - 6);
145
146 bool valid = m_policy->AuthorizeWithCert(nodeId, role, data);
147 ZtLogger::LogCertValidationResult(nodeId, valid);
148 m_authorized[nodeId] = valid;
149
150 if (valid) {
151 m_handshake->StartHandshake(GetNode(), GetNode(), nodeId, 1);
152 }
153 } catch (...) {
154 ZtLogger::Log("ZT", "Malformed certificate – ignored");
155 }
156 continue;
157 }
158
159 bool validPacket = false;
160 for (const auto& entry : m_authorized) {
161 if (entry.second && m_handshake->HasSession(entry.first)) {
162 try {
163 std::string sessionKey = m_handshake->GetSessionKey(entry.first);
164 std::vector<byte> rawKey = HexToBytes(sessionKey);
165 std::string decrypted = DecryptPayload(data, rawKey.data());
166 ZtLogger::LogDecryption(decrypted);
167 validPacket = true;
168 break;
169 } catch (...) {
170 ZtLogger::LogDecryptionFailure();
171 break;
172 }
173 }
174 }
175
176 if (!validPacket) {
177 ZtLogger::Log("ZT", "Unauthorized data attempt - dropped");
178 }
179 }
