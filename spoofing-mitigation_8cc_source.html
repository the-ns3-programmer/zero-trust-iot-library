125 // Certificate validation
126 if (data.find("ID:") == 0) {
127 size_t idPos = data.find("ID:");
128 size_t rolePos = data.find("|ROLE:");
129 size_t expPos = data.find("|EXP:");
130 uint32_t nodeId = std::stoul(data.substr(idPos + 3, rolePos - idPos - 3));
131 std::string role = data.substr(rolePos + 6, expPos - rolePos - 6);
132
133 bool valid = zt->AuthorizeWithCert(nodeId, role, data);
134 ZtLogger::LogCertValidationResult(nodeId, valid);
135 authorized[nodeId] = valid;
136
137 handshake->StartHandshake(GetNode(), GetNode(), nodeId, 1);
138 return;
139 }
140
141 // Decryption only if authorized
142 uint32_t peerId = 0;
143 if (authorized[peerId]) {
144 try {
145 std::string sessionHexKey = handshake->GetSessionKey(1);
146 std::vector<byte> rawKey = HexToBytes(sessionHexKey);
147 std::string decrypted = DecryptPayload(data, rawKey.data());
148 ZtLogger::LogDecryption(decrypted);
149 } catch (...) {
150 ZtLogger::LogDecryptionFailure();
151 }
152 } else {
153 ZtLogger::Log("ZT", "Unauthorized data attempt");
